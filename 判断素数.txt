#include<iostream>
#include<math.h>
using namespace std;
int i;
class Prime
{
public:
	Prime();
	void cal();
	~Prime();
private:
	int num;
	int f;
};
Prime::Prime()
{
	cin>>num;
}
void Prime::cal()
{
	for(i=2 ; i<=num ; i++)
		if(num%i == 0)
			break;
	if(i == num)
		f = 1;
	else
		f = 0;
}
	
Prime::~Prime()
{
	if(f == 1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
}
int main()
{
	Prime p;
	p.cal();
	return 0;
}
