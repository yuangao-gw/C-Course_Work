#include<iostream.h>
class String
{
public:
	String(char *e);
	String();
	bool operator!();
	void dispaly();
private:
	char *elems;
};
String::String(char *e)
{
	elems = e;
}
String::String()
{
	elems = NULL;
}
bool String::operator !()
{
	if(elems == NULL)
		return true;
	else 
		return false;
}
void String::dispaly()
{
	if(elems != NULL)
	cout<<elems;
}
int main()
{
	String str;//("I am a Student\n");
	str.dispaly();
	if(!str)
		cout<<"这是一个空字符串！"<<endl;
	else
		cout<<"这不是一个空字符串！"<<endl;
	return 0;
}
