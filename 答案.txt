    int bit;  
    int j = 0;  
    long long end=0;  
    long long mod = 0;  
    long long temp = 0;  
    for (int i =1; i<10000; i++)  
    {  
        lng = 1;  
        bit = 1;  
        end = 0;  
        mod = 0;  
        j = 0;  
  
        temp = i;  
        lng = (long long )pow(temp, 3); //pow( )函数的返回值类型为double，所以强制转换为long long 类型  
  
        while (temp >0)  //判断i是几位数，取余的时候则循环几次。  
        {  
            ++j;    //求i是几位数，取余的时候共参考。  
            temp = temp / 10;  
        }  
  
        while (j-- > 0)  
        {  
            bit = bit * 10;  
        }  
        end = lng%bit;  //根据i的位数，来判断取余的时候取几位。  
        if (end == i)  
        {  
            count++;  
            cout << i << endl;  //用于调试，让用户看的更直接。  
        }  
    }  
    cout << count << endl;  
    system("pause");

}  s
