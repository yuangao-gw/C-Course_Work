#include<iostream>
#include<math.h>
using namespace std;
template<typename T>
class Complex 
{
private:
	T x;
	T y;
public:
	Complex (T a, T b):x(a),y(b){}
	int max()
	{
		return x>y?x:y;
	}
	int min()
			{
		return x<y?x:y;
	}
};
int main()
{
	Complex<int> i(10,5);
	cout<<"最大值"<<i.max()<<endl;
	cout<<"最小值"<<i.min()<<endl;
	return 0;
}
