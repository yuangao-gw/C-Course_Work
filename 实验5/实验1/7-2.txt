#include<iostream>
using namespace std;
template<typename T>
T max(T a,T b, T c)
{
	T m;
	m = a>b?a:b;
	m = m>c?m:c;
	return m;
}
int main()
{
	{
		int x, y, z;
		cin>>x>>y>>z;
		cout<<"3个 int 型数的最大值为："<<max(x,y,z)<<endl;
	}
	{
		float x, y, z;
		cin>>x>>y>>z;
		cout<<"3个 float 型数的最大值为："<<max(x,y,z)<<endl;
	}
	{
		double x, y, z;
		cin>>x>>y>>z;
		cout<<"3个 double 型数的最大值为："<<max(x,y,z)<<endl;
	}

	return 0;
}
