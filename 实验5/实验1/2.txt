#include<iostream>
using namespace std;
int main()
{
	int i, j, t,a[10], sum;
	float v;
	cout<<"排序前:"<<endl;
	for(i=0;i<5;i++)
	{
		cout<<"arr["<<i+1<<"]  = ";
		cin>>a[i];
	}
	for(i=0;i<4;i++)
		for(j=i+1;j<5;j++)
			if(a[i]<a[j])
			{
				t = a[i];
				a[i] = a[j];
				a[j] = t;
			}
	cout<<"排序后:"<<endl;
	for(i=0;i<5;i++)
    cout<<"arr["<<i+1<<"]  = "<<a[i]<<endl;
	sum=0;
    for(i=0;i<5;i++) 
	sum = sum + a[i];
	v = sum/5.0;
	cout<<"数组平均值为:"<<v<<endl;;
	return 0;
}
