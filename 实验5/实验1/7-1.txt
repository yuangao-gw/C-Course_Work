#include<iostream>
using namespace std;
int max(int a,int b,int c)
{
	int m;
	m=a>b?a:b;
	m=m>c?m:c;
	return m;
}
float max(float a,float b,float c)
{
	float m;
	m=a>b?a:b;
	m=m>c?m:c;
	return m;
}
double max(double a,double b,double c)
{
	double m;
	m=a>b?a:b;
	m=m>c?m:c;
	return m;
}
int main()
{
	{
		int x,y,z;
		cin>>x>>y>>z;
		cout<<"3个  int  型数的最大值为："<<max(x,y,z)<<endl;
	}
	{
		double x,y,z;
		cin>>x>>y>>z;
		cout<<"3个  int  型数的最大值为："<<max(x,y,z)<<endl;
	}
	{
		float x,y,z;
		cin>>x>>y>>z;
		cout<<"3个  int  型数的最大值为："<<max(x,y,z)<<endl;
	}
		return 0;
}
