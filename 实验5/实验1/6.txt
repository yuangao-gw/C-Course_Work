#include<iostream>
using namespace std;
int max1(int *a,int *b)
{
	return *a>*b?*a:*b;
}
int max2(int &a, int &b)
{
	return a>b?a:b;
}
int main()
{
	int n,m;
	cout<<"请输入两个数：";
	cin>>n>>m;
	cout<<"形参使用指针，比较后较大的数是: "<<max1(&n,&m)<<endl;
	cout<<"形参使用引用，比较后较大的数是: "<<max2(n,m)<<endl;
	return 0;
}
