#include<iostream>
#include<string>
using namespace std;
double max(double a=8.6,double b=7.4,double c=4.5)
{
	double m;
	m=a>b?a:b;
	m=m>c?m:c;
	return m;
}
int main()
{
	double x,y,z;
	cout<<"3个缺省参数分别为：8.6 7.4 4.5"<<endl;
	cout<<"请输入3个数："<<endl;
	cin>>x>>y>>z;
	cout<<"最大值为"<<max(x,y,z)<<endl;

	cout<<"请输入两个数："<<endl;
	cin>>x>>y;
	cout<<"一个参数缺省后，最大值为："<<max(x,y)<<endl;

		cout<<"请输入一个数："<<endl;
	cin>>x;
	cout<<"两个参数缺省后，最大值为："<<max(x)<<endl;
	cout<<"3个参数缺省后，最大值为："<<max()<<endl;
	return 0;
}
