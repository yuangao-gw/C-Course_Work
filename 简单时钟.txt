#include<iostream>
#include<cstdlib>
#include<windows.h>
using namespace std;
class Time
{
private:
	int hour;
	int minute;
	int sec;
public:
	Time();
	void set_time();
	void show_time();
};
Time::Time()
{
	hour = 0;
	minute = 0;
	sec = 0;

	while(1)
	{
		show_time();
		sec++;
		minute = minute+sec/60;
		sec=sec%60;
		hour = hour+minute/60;
		minute=minute%60;
		hour = hour/24;
		Sleep(1000);
		system("cls");
	}
}
void Time::set_time()
{
	cin>>hour;
	cin>>minute;
	cin>>sec;
}
void Time::show_time()
{
	cout<<hour <<":"<<minute <<":"<<sec <<endl;
}
int main()
{
	Time t;
	return 0;
}
