#include<iostream>
#include<string>
using namespace std;
int n, i;
class Max	
{
private:
	int a[1010];
	int max;
	int n;
public:
	void setdata();
	void display();
	void cul();
};
void Max::setdata()
{
	cin>>n;
	for(i = 0; i < n; i++)
    	cin>>a[i];
}
void Max::cul()
{
	max = a[0];
	for(i = 0; i < n; i++)
		if(a[i] > max)
			max = a[i];
}
void Max::display()
{
	cout<<"max = "<<max<<endl;
}
int main() 
{

	Max t;
	t.setdata();
	t.cul();
	t.display();
	return 0;
}
