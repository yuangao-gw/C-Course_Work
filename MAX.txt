#include<iostream>
using namespace std;
int main()
{
	int max(int a, int b);
	int max(int a, int b, int c);
	int max(int a, int b, int c, int d);
	int a, b, c, d;
	while(cin>>a>>b>>c>>d)
	{
		cout<<max(a,b)<<"\n";
		cout<<max(a,b,c)<<endl;
		cout<<max(a,b,c,d)<<endl;
	}
	return 0;
}
int max(int a, int b)
{
	int m;
	m = a>b?a:b;
	return m;
}

int max(int a, int b, int c)
{
	int m;
	m = a>b?a:b;
	m = m>c?m:c;
	return m;
}
int max(int a, int b, int c, int d)
{
	int m;
	m = a>b?a:b;
	m = m>c?m:c;
    m = m>d?m:d;
	return m;
}
