#include<iostream.h>
#include<string.h>
class Student;
class Teacher
{
private:
	long num;
	char name[20];
	char sex;
public:
	Teacher operator = (Student s);
	friend ostream & operator << (ostream & output, Teacher &t);
};
class Student
{
private:
	long num;
	char name[20];
	char sex;
public:
	Student(long n,char na[], char s)
	{
		num = n; 
		strcpy(name, na);
		sex = s;
	}
	friend Teacher Teacher::operator = (Student s);
};
ostream & operator << (ostream & output, Teacher &t)
{
	output<<"工号："<<t.num<<endl;
	output<<"姓名："<<t.name<<endl;
	output<<"性别："<<t.sex<<endl;
	return output;
}
Teacher Teacher::operator = (Student s)
{
	num = s.num;
	strcpy(name, s.name);
	sex = s.sex;
	return *this;
}
int main()
{
	Student s(2015110,"zhangsan",'f');
	Teacher t;
	t = s;
	cout<<t;
	return 0;
}
